{#---------------------------------------------------------------------------#}
{# Macros                                                                    #}
{#---------------------------------------------------------------------------#}

{%- macro render_location(event) -%}
    {%- if event.location -%}
        {%- if event.room -%}
            {{ event.location | trim }} {{ render_room(event) }}.
        {%- else -%}
            {{ event.location | trim }}.
        {%- endif -%}
        {{- render_address(event) }}
    {%- elif event.room -%}
        {{ render_room(event) }}.
        {{- render_address(event) }}
    {%- else -%}
        {{ render_address(event) }}
    {%- endif -%}
{%- endmacro -%}

{%- macro render_room(event) -%}
    {%- if event.room -%}
        {%- if event.location -%}
                ({{ event.room | trim }})
        {%- else -%}
            {%- trans room=event.room | trim | ensure_unicode -%}
                room {{ room }}
            {%- endtrans -%}
        {%- endif -%}
    {%- endif -%}
{%- endmacro -%}

{%- macro render_address(event) -%}
    {%- if event.address -%}
        {%- if event.location or event.room -%}
            {%- trans %}
Address:
            {%- endtrans -%}
        {%- else -%}
            {%- trans -%}
the following address:
            {%- endtrans -%}
        {%- endif %}

    {{ event.address | trim | indent }}
    {%- endif -%}
{%- endmacro -%}
